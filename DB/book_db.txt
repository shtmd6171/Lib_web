CREATE DATABASE lib;

CREATE TABLE returned (
returned_id int(20) NOT NULL,
returned_date date NOT NULL,
book_id int(20) NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE loan (
book_id int(20) NOT NULL,
user_id int(20) NOT NULL,
loan_date date NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE user (
user_id int(20) NOT NULL,
pwd varchar(60) NOT NULL,
email varchar(20) NOT NULL,
addr varchar(30) NOT NULL,
name varchar(15) NOT NULL,
gender varchar(15) NOT NULL,
tel varchar(15) NOT NULL,
code varchar(5) NOT NULL DEFAULT 'U' ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE book (
book_id int(20) NOT NULL,
title varchar(50) NOT NULL,
author varchar(20) NOT NULL,
publisher varchar(20) NOT NULL,
the_date date NOT NULL,
genre varchar(20) NOT NULL,
file varchar(60) NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE book_review (
review_id int(20) NOT NULL,
review_title varchar(20) NOT NULL,
review_desc varchar(100) NOT NULL,
book_id int(20) NOT NULL,
user_id int(20) NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE comment (
comment_id int(20) NOT NULL,
comm_description varchar(80) NOT NULL,
review_id int(20) NOT NULL,
user_id int(20) NOT NULL ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
